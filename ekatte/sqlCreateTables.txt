DROP TABLE selishte;
DROP TABLE obstina;
DROP TABLE oblast;
DROP TABLE helperselishte;
DROP TABLE helperobstina;
DROP TABLE helperoblast;

CREATE TABLE public.oblasti
(
  id text,
  oblast_kod text,
  ekatte text,
  name text,
  region text,
  document text,
  abc text,
  CONSTRAINT oblasti_pkey PRIMARY KEY (oblast_kod)
)
WITH (
  OIDS=FALSE
);
ALTER TABLE public.oblasti
  OWNER TO postgres;


CREATE TABLE public.obstini
(
  obstina_kod text,
  ekatte text,
  name text,
  category text,
  document text,
  abc text,
  oblast text,
  CONSTRAINT obstini_pkey PRIMARY KEY (obstina_kod),
  FOREIGN KEY (oblast) REFERENCES oblasti (oblast_kod)
)
WITH (
  OIDS=FALSE
);
ALTER TABLE public.obstini
  OWNER TO postgres;
  

CREATE TABLE public.selishta
(
  ekatte text,
  t_v_m text,
  name text,
  oblast_kod text,
  obstina_kod text,
  kmetstvo_kod text,
  kind text,
  category text,
  altitude text,
  document text,
  tsb text,
  abc text,
  CONSTRAINT selishta_pkey PRIMARY KEY (ekatte),
  FOREIGN KEY (obstina_kod) REFERENCES obstini (obstina_kod)
)
WITH (
  OIDS=FALSE
);
ALTER TABLE public.selishta
  OWNER TO postgres;
