CREATE TABLE "monitors" (
	"inches" TEXT NOT NULL,
	"resolution" TEXT NOT NULL,
	"makerid" integer NOT NULL,
	"price" TEXT NOT NULL,
	"model" serial NOT NULL,
	"amount" serial NOT NULL,
	CONSTRAINT monitors_pk PRIMARY KEY ("model")
) WITH (
  OIDS=FALSE
);

	"makerid" integer NOT NULL,
	"model" TEXT NOT NULL,
	"amount" TEXT NOT NULL,
	CONSTRAINT keyboards_pk PRIMARY KEY ("model")
) WITH (
  OIDS=FALSE
);

	"makerid" integer NOT NULL,
	"model" TEXT NOT NULL,
	"price" TEXT NOT NULL,
	"amount" TEXT NOT NULL,
	"storage" TEXT NOT NULL,
	CONSTRAINT hdds_pk PRIMARY KEY ("model")
) WITH (
  OIDS=FALSE
);

  OIDS=FALSE

CREATE TABLE "soundcards" (
	"makerid" integer NOT NULL,

	"price" TEXT NOT NULL,
	"amount" TEXT NOT NULL,

) WITH (
  OIDS=FALSE

	"name" TEXT NOT NULL,
